calc_adj_r2 <- function(r2, n, p) {
    if (is.na(r2) || n <= p + 1) return(NA_real_)
    1 - (1 - r2) * (n - 1) / (n - p - 1)
}

safe_polr_multi <- function(formula, data) {
    tryCatch({
        mf <- model.frame(formula, data)
        y <- mf[[1]]
        x <- mf[, -1, drop = FALSE]
        if (length(unique(y)) < 2) return(rep(0.333, ncol(x)))
        if (any(apply(x, 2, sd, na.rm = TRUE) == 0)) return(rep(0.333, ncol(x)))
        if (any(table(y) < 3)) return(rep(0.333, ncol(x)))
        m <- polr(formula, data = data, Hess = TRUE)
        coefs <- coef(m)
        w <- abs(coefs) / sum(abs(coefs))
        return(w)
    }, error = function(e) {
        return(rep(0.333, 3))
    })
}

    if (b %% 20 == 0) cat("  Bootstrap:", b, "/", n_boot, "\n")
    
    train_idx <- sample(1:length(target), size = floor(0.8 * length(target)), replace = TRUE)
    test_idx <- setdiff(1:length(target), unique(train_idx))
    if (length(test_idx) < 5) next
    
    X_train <- cbind(features_micro[train_idx, ], cov_mat[train_idx, ])
    X_test  <- cbind(features_micro[test_idx, ], cov_mat[test_idx, ])
    
    model <- xgb.train(
        params = params,
        data = xgb.DMatrix(data = X_train, label = target[train_idx]),
        nrounds = 200,
        verbose = 0
    )
    
    pred_train <- predict(model, xgb.DMatrix(data = X_train))
    pred_test  <- predict(model, xgb.DMatrix(data = X_test))
    
    y_tr <- target[train_idx]
    y_te <- target[test_idx]
    n_te <- length(y_te)
    n_tr <- length(train_idx)
    p <- ncol(X_train)
    
    r2_tr <- 1 - sum((y_tr - pred_train)^2) / sum((y_tr - mean(y_tr))^2)
    r2_te <- 1 - sum((y_te - pred_test)^2) / sum((y_te - mean(y_te))^2)
    adj_r2_te <- calc_adj_r2(r2_te, n_tr, p)
    rmse_te <- sqrt(mean((y_te - pred_test)^2))
    spearman_te <- cor(pred_test, y_te, method = "spearman")
    
    boot_res <- rbind(boot_res, data.frame(
        Bootstrap = b, Train_R2 = r2_tr, Test_R2 = r2_te, 
        Adj_R2 = adj_r2_te, RMSE = rmse_te, Spearman = spearman_te,
        Test_N = n_te
    ))
    
    imp <- xgb.importance(model = model, feature_names = colnames(X_train))
    boot_imp[[b]] <- imp
}

cv_eval <- function(X, y, covs, nfolds = 5) {
    set.seed(123)
    folds <- createFolds(y, k = nfolds, list = TRUE)
    out <- data.frame(R2 = numeric(nfolds), Adj_R2 = numeric(nfolds), RMSE = numeric(nfolds))
    for (i in 1:nfolds) {
        te <- folds[[i]]
        tr <- setdiff(1:length(y), te)
        if (is.null(X)) {
            X_tr <- covs[tr, , drop = FALSE]
            X_te <- covs[te, , drop = FALSE]
        } else {
            X_tr <- cbind(X[tr, ], covs[tr, ])
            X_te <- cbind(X[te, ], covs[te, ])
        }
        dtr <- xgb.DMatrix(data = X_tr, label = y[tr])
        dte <- xgb.DMatrix(data = X_te, label = y[te])
        m <- xgb.train(params = list(objective = "reg:squarederror", eval_metric = "rmse",
                                     eta = 0.05, max_depth = 4, subsample = 0.8, colsample_bytree = 0.8),
                       data = dtr, nrounds = 200, evals = list(test = dte),
                       early_stopping_rounds = 10, verbose = 0)
        pr <- predict(m, dte)
        n_tr <- length(tr)
        p <- ncol(X_tr)
        r2 <- 1 - sum((y[te] - pr)^2) / sum((y[te] - mean(y[te]))^2)
        adj_r2 <- calc_adj_r2(r2, n_tr, p)
        rmse <- sqrt(mean((y[te] - pr)^2))
        out[i, ] <- c(r2, adj_r2, rmse)
    }
    colMeans(out)
}

perm_test_micro_detailed <- function(X_micro, y, covs, n_perm = 100) {
    set.seed(999)
    train_idx <- sample(1:length(y), size = floor(0.8 * length(y)))
    test_idx <- setdiff(1:length(y), train_idx)
    y_te <- y[test_idx]
    y_tr <- y[train_idx]
    
    # 真实模型
    X_tr_micro <- cbind(X_micro[train_idx, ], covs[train_idx, ])
    X_te_micro <- cbind(X_micro[test_idx, ], covs[test_idx, ])
    m_micro <- xgb.train(params = list(objective = "reg:squarederror", eval_metric = "rmse",
                                       eta = 0.05, max_depth = 4, subsample = 0.8, colsample_bytree = 0.8),
                         data = xgb.DMatrix(data = X_tr_micro, label = y_tr), nrounds = 200, verbose = 0)
    pred_micro <- predict(m_micro, xgb.DMatrix(data = X_te_micro))
    r2_micro <- 1 - sum((y_te - pred_micro)^2) / sum((y_te - mean(y_te))^2)
    
    imp_true <- xgb.importance(model = m_micro, feature_names = colnames(X_tr_micro))
    imp_true <- imp_true[!imp_true$Feature %in% c("Age", "Gender_num", "BMI"), ]
    
    # Baseline
    X_tr_base <- covs[train_idx, ]
    X_te_base <- covs[test_idx, ]
    m_base <- xgb.train(params = list(objective = "reg:squarederror", eval_metric = "rmse",
                                      eta = 0.05, max_depth = 4, subsample = 0.8, colsample_bytree = 0.8),
                        data = xgb.DMatrix(data = X_tr_base, label = y_tr), nrounds = 200, verbose = 0)
    pred_base <- predict(m_base, xgb.DMatrix(data = X_te_base))
    r2_base <- 1 - sum((y_te - pred_base)^2) / sum((y_te - mean(y_te))^2)
    
    delta_true <- r2_micro - r2_base
    
    all_species <- colnames(X_micro)
    n_species <- length(all_species)
    perm_imp_matrix <- matrix(0, nrow = n_species, ncol = n_perm)
    rownames(perm_imp_matrix) <- all_species
    perm_delta <- numeric(n_perm)
    
    cat("  Running", n_perm, "permutations...\n")
    
    for (i in 1:n_perm) {
        if (i %% 20 == 0) cat("    Permutation", i, "/", n_perm, "\n")
        
        perm_idx <- sample(nrow(X_micro))
        X_perm <- X_micro[perm_idx, ]
        
        X_tr_p <- cbind(X_perm[train_idx, ], covs[train_idx, ])
        X_te_p <- cbind(X_perm[test_idx, ], covs[test_idx, ])
        m_p <- xgb.train(params = list(objective = "reg:squarederror", eval_metric = "rmse",
                                       eta = 0.05, max_depth = 4, subsample = 0.8, colsample_bytree = 0.8),
                         data = xgb.DMatrix(data = X_tr_p, label = y_tr), nrounds = 200, verbose = 0)
        
        pred_p <- predict(m_p, xgb.DMatrix(data = X_te_p))
        r2_p <- 1 - sum((y_te - pred_p)^2) / sum((y_te - mean(y_te))^2)
        perm_delta[i] <- r2_p - r2_base
        
        imp_p <- xgb.importance(model = m_p, feature_names = colnames(X_tr_p))
        imp_p <- imp_p[!imp_p$Feature %in% c("Age", "Gender_num", "BMI"), ]
        if (nrow(imp_p) > 0) {
            match_idx <- match(imp_p$Feature, all_species)
            valid <- !is.na(match_idx)
            perm_imp_matrix[match_idx[valid], i] <- imp_p$Gain[valid]
        }
    }
    
    true_gain_vec <- numeric(n_species)
    names(true_gain_vec) <- all_species
    match_true <- match(imp_true$Feature, all_species)
    valid_true <- !is.na(match_true)
    true_gain_vec[match_true[valid_true]] <- imp_true$Gain[valid_true]
    
    species_perm_stats <- data.frame(
        Species = all_species,
        True_Gain = true_gain_vec,
        Perm_Mean_Gain = rowMeans(perm_imp_matrix),
        Perm_SD_Gain = apply(perm_imp_matrix, 1, sd),
        Perm_p_value = sapply(1:n_species, function(j) {
            if (true_gain_vec[j] == 0) return(1.0)
            mean(perm_imp_matrix[j, ] >= true_gain_vec[j])
        }),
        stringsAsFactors = FALSE
    )
    
    overall_p <- mean(perm_delta >= delta_true)
    cat("  Overall DeltaR2 =", round(delta_true, 4), ", p =", round(overall_p, 4), "\n")
    cat("  Species with Perm_p < 0.05:", sum(species_perm_stats$Perm_p_value < 0.05), "/", n_species, "\n")
    
    list(
        delta_true = delta_true, r2_micro = r2_micro, r2_base = r2_base, 
        perm_delta = perm_delta, overall_p = overall_p,
        species_perm_stats = species_perm_stats
    )
}

# 12. WEIGHT STABILITY (M5)
# ==============================================================================
cat("\n--- Weight Stability (5-fold CV) ---\n")
weight_mat <- matrix(NA, nrow = 5, ncol = 3)
colnames(weight_mat) <- c("XGBoost", "Monotonicity", "Correlation")

set.seed(789)
folds_w <- createFolds(target, k = 5, list = TRUE)
for (i in 1:5) {
    tr_idx <- setdiff(1:length(target), folds_w[[i]])
    
    tr_scores_list <- lapply(tr_idx, function(j) {
        w <- raw_features[j, ] + 1e-6
        data.frame(
            Score_XGBoost = weighted.mean(score_xgb_vec, w = w, na.rm = TRUE),
            Score_Monotonicity = weighted.mean(score_mono_vec, w = w, na.rm = TRUE),
            Score_Correlation = weighted.mean(score_corr_vec, w = w, na.rm = TRUE)
        )
    })
    tr_scores_df <- do.call(rbind, tr_scores_list)
    
    tr_data <- data.frame(
        group = factor(species_data_sub$group[tr_idx], levels = c("HC", "ANIH", "AIH", "ALF"), ordered = TRUE),
        Score_XGBoost = as.numeric(scale(tr_scores_df$Score_XGBoost)),
        Score_Monotonicity = as.numeric(scale(tr_scores_df$Score_Monotonicity)),
        Score_Correlation = as.numeric(scale(tr_scores_df$Score_Correlation))
    )
    
    w_fit <- safe_polr_multi(group ~ Score_XGBoost + Score_Monotonicity + Score_Correlation, 
                             data = tr_data)
    weight_mat[i, ] <- w_fit
}
cat("Weight stability across 5 folds:\n")
print(round(weight_mat, 4))
cat("Mean (SD):", paste(round(colMeans(weight_mat), 3), "±", round(apply(weight_mat, 2, sd), 3)), "\n")
